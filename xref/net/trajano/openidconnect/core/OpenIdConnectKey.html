3   /**
4    * Registered Open ID Connect Keys. This unifies all the key string constants
5    * defined in OpenID Connect specifications.
6    *
7    * @author Archimedes
8    */
9   public final class OpenIdConnectKey {
10  
11      /**
12       * OPTIONAL. Requested Authentication Context Class Reference values.
13       * Space-separated string that specifies the acr values that the
14       * Authorization Server is being requested to use for processing this
15       * Authentication Request, with the values appearing in order of preference.
16       * The Authentication Context Class satisfied by the authentication
17       * performed is returned as the acr Claim Value, as specified in Section 2.
18       * The acr Claim is requested as a Voluntary Claim by this parameter.
19       */
20      public static final String ACR_VALUES = "acr_values";
21  
22      /**
23       * REQUIRED. OAuth 2.0 Client Identifier valid at the Authorization Server.
24       */
25      public static final String CLIENT_ID = "client_id";
26  
27      public static final String CLIENT_SECRET = "client_secret";
28      public static final String CLAIMS = "claims";
29  
30      public static final String CODE = "code";
31  
32      /**
33       * OPTIONAL. ASCII string value that specifies how the Authorization Server
34       * displays the authentication and consent user interface pages to the
35       * End-User. The defined values are:
36       */
37      public static final String DISPLAY = "display";
38  
39      /**
40       * Error code. Possible values are in {@link ErrorCode}.
41       */
42      public static final String ERROR = "error";
43  
44      /**
45       * OPTIONAL. Human-readable ASCII encoded text description of the error.
46       */
47      public static final String ERROR_DESCRIPTION = "error_description";
48  
49      /**
50       * OPTIONAL. URI of a web page that includes additional information about
51       * the error.
52       */
53      public static final String ERROR_URI = "error_uri";
54  
55      public static final String GRANT_TYPE = "grant_type";
56  
57      /**
58       * <p>
59       * OPTIONAL. ID Token previously issued by the Authorization Server being
60       * passed as a hint about the End-User's current or past authenticated
61       * session with the Client. If the End-User identified by the ID Token is
62       * logged in or is logged in by the request, then the Authorization Server
63       * returns a positive response; otherwise, it SHOULD return an error, such
64       * as login_required. When possible, an id_token_hint SHOULD be present when
65       * prompt=none is used and an invalid_request error MAY be returned if it is
66       * not; however, the server SHOULD respond successfully when possible, even
67       * if it is not present. The Authorization Server need not be listed as an
68       * audience of the ID Token when it is used as an id_token_hint value.
69       * </p>
70       * <p>
71       * If the ID Token received by the RP from the OP is encrypted, to use it as
72       * an id_token_hint, the Client MUST decrypt the signed ID Token contained
73       * within the encrypted ID Token. The Client MAY re-encrypt the signed ID
74       * token to the Authentication Server using a key that enables the server to
75       * decrypt the ID Token, and use the re-encrypted ID token as the
76       * id_token_hint value.
77       * </p>
78       */
79      public static final String ID_TOKEN_HINT = "id_token_hint";
80  
81      /**
82       * OPTIONAL. Hint to the Authorization Server about the login identifier the
83       * End-User might use to log in (if necessary). This hint can be used by an
84       * RP if it first asks the End-User for their e-mail address (or other
85       * identifier) and then wants to pass that value as a hint to the discovered
86       * authorization service. It is RECOMMENDED that the hint value match the
87       * value used for discovery. This value MAY also be a phone number in the
88       * format specified for the phone_number Claim. The use of this parameter is
89       * left to the OP's discretion.
90       */
91      public final static String LOGIN_HINT = "login_hint";
92  
93      /**
94       * OPTIONAL. Maximum Authentication Age. Specifies the allowable elapsed
95       * time in seconds since the last time the End-User was actively
96       * authenticated by the OP. If the elapsed time is greater than this value,
97       * the OP MUST attempt to actively re-authenticate the End-User. (The
98       * max_age request parameter corresponds to the OpenID 2.0 PAPE
99       * [OpenID.PAPE] max_auth_age request parameter.) When max_age is used, the
100      * ID Token returned MUST include an auth_time Claim Value.
101      */
102     public static final String MAX_AGE = "max_age";
103 
104     /**
105      * OPTIONAL. String value used to associate a Client session with an ID
106      * Token, and to mitigate replay attacks. The value is passed through
107      * unmodified from the Authentication Request to the ID Token. Sufficient
108      * entropy MUST be present in the nonce values used to prevent attackers
109      * from guessing values. For implementation notes, see Section 15.5.2.
110      */
111     public static final String NONCE = "nonce";
112 
113     /**
114      * OPTIONAL. Space delimited, case sensitive list of ASCII string values
115      * that specifies whether the Authorization Server prompts the End-User for
116      * reauthentication and consent. The defined values are:
117      */
118     public static final String PROMPT = "prompt";
119 
120     /**
121      * REQUIRED. Redirection URI to which the response will be sent. This URI
122      * MUST exactly match one of the Redirection URI values for the Client
123      * pre-registered at the OpenID Provider, with the matching performed as
124      * described in Section 6.2.1 of [RFC3986] (Simple String Comparison). When
125      * using this flow, the Redirection URI SHOULD use the https scheme;
126      * however, it MAY use the http scheme, provided that the Client Type is
127      * confidential, as defined in Section 2.1 of OAuth 2.0, and provided the OP
128      * allows the use of http Redirection URIs in this case. The Redirection URI
129      * MAY use an alternate scheme, such as one that is intended to identify a
130      * callback into a native application.
131      */
132     public static final String REDIRECT_URI = "redirect_uri";
133 
134     public static final String REQUEST = "request";
135 
136     /**
137      * OPTIONAL. Informs the Authorization Server of the mechanism to be used
138      * for returning parameters from the Authorization Endpoint. This use of
139      * this parameter is NOT RECOMMENDED when the Response Mode that would be
140      * requested is the default mode specified for the Response Type.
141      */
142     public static final String RESPONSE_MODE = "response_mode";
143 
144     /**
145      * REQUIRED. OAuth 2.0 Response Type value that determines the authorization
146      * processing flow to be used, including what parameters are returned from
147      * the endpoints used. When using the Authorization Code Flow, this value is
148      * code.
149      */
150     public static final String RESPONSE_TYPE = "response_type";
151 
152     /**
153      * REQUIRED. OpenID Connect requests MUST contain the openid scope value. If
154      * the openid scope value is not present, the behavior is entirely
155      * unspecified. Other scope values MAY be present. Scope values used that
156      * are not understood by an implementation SHOULD be ignored. See Sections
157      * 5.4 and 11 for additional scope values defined by this specification.
158      */
159     public static final String SCOPE = "scope";
160 
161     /**
162      * An opaque value used to maintain state between the request and the
163      * callback. Typically, Cross-Site Request Forgery (CSRF, XSRF) mitigation
164      * is done by cryptographically binding the value of this parameter with a
165      * browser cookie. This is REQUIRED if the Authorization Request included
166      * the state parameter. Set to the value received from the Client.
167      */
168     public static final String STATE = "state";
169 
170     /**
171      * OPTIONAL. End-User's preferred languages and scripts for the user
172      * interface, represented as a space-separated list of BCP47 [RFC5646]
173      * language tag values, ordered by preference. For instance, the value
174      * "fr-CA fr en" represents a preference for French as spoken in Canada,
175      * then French (without a region designation), followed by English (without
176      * a region designation). An error SHOULD NOT result if some or all of the
177      * requested locales are not supported by the OpenID Provider.
178      */
179     public static final String UI_LOCALES = "ui_locales";
180 
181     public static final String REFRESH_TOKEN = "refresh_token";
182

184 
185     }
