3   /**
4    * <p>
5    * <a href="http://tools.ietf.org/html/rfc6749#section-4.1.3">Access Token
6    * Request</a>.
7    * </p>
8    * <p>
9    * The client makes a request to the token endpoint by sending the following
10   * parameters using the "application/x-www-form-urlencoded" format per Appendix
11   * B with a character encoding of UTF-8 in the HTTP request entity-body. This is
12   * used in
13   * <a href="http://openid.net/specs/openid-connect-core-1_0.html#TokenRequest">
14   * OpenID Token Requests</a>.
15   * </p>
16   */
17  public final class TokenRequestParam {
18  
19      /**
20       * REQUIRED, if the client is not authenticating with the authorization
21       * server as described in Section 3.2.1.
22       */
23      public static final String CLIENT_ID = "client_id";
24  
25      /**
26       * REQUIRED. The authorization code received from the authorization server.
27       */
28      public static final String CODE = "code";
29  
30      /**
31       * REQUIRED. Value MUST be set to "authorization_code".
32       */
33      public static final String GRANT_TYPE = "grant_type";
34  
35      /**
36       * REQUIRED, if the "redirect_uri" parameter was included in the
37       * authorization request as described in Section 4.1.1, and their values
38       * MUST be identical.
39       */
40      public static final String REDIRECT_URI = "redirect_uri";
41  
42  }
