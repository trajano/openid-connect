3   import javax.xml.bind.annotation.XmlEnum;
4   import javax.xml.bind.annotation.XmlEnumValue;
5   
6   /**
7    * Subject identifier types. A Subject Identifier is a locally unique and never
8    * reassigned identifier within the Issuer for the End-User, which is intended
9    * to be consumed by the Client. Two Subject Identifier types are defined by
10   * this specification.: {@link #PUBLIC} and {@link #PAIRWISE}. The OpenID
11   * Provider's Discovery document SHOULD list its supported Subject Identifier
12   * types in the subject_types_supported element. If there is more than one type
13   * listed in the array, the Client MAY elect to provide its preferred identifier
14   * type using the subject_type parameter during Registration.
15   *
16   * @author Archimedes
17   */
18  @XmlEnum
19  public enum SubjectIdentifierType {
20      /**
21       * This provides a different sub value to each Client, so as not to enable
22       * Clients to correlate the End-User's activities without permission.
23       */
24      @XmlEnumValue("pairwise") PAIRWISE,
25      /**
26       * This provides the same sub (subject) value to all Clients. It is the
27       * default if the provider has no subject_types_supported element in its
28       * discovery document.
29       */
30      @XmlEnumValue("public") PUBLIC;
31  }
